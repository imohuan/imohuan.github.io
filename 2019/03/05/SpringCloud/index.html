Spring Cloud

注册 Eureka Server端
- Application
@EnableEurekaServer    
public class Demo1Application {}

- Application.yml
spring:
  application:
    name: eureka
server:
  port: 8671

eureka:
  client:
    register-with-eureka: false
    service-url:
      defaultZone: http://localhost:${server.port}/${spring.application.name}
  server:
    # 客户端上线 下线 自我保护   （开发阶段使用）   只在server端注册
    enable-self-preservation: false
注册 Eureka Clint端
- Application
@EnableEurekaClient
public class Demo1Application {}

- Application.yml
spring:
  application:
    name: product
eureka:
  client:
    service-url:
      defaultZone: http://localhost:8671/eureka        // 注册到上面的server中

Feign
<dependency>
    <groupId>org.springframework.cloud</groupId>
    <artifactId>spring-cloud-starter-feign</artifactId>
</dependency>

- 启用注解
@EnableFeignClients

- 定义需要的服务
@FeignClient( name = "product" )
public interface ProductClient {
    @GetMapping("/msg')
     String productMsg();     // 作为客服端访问 
}

- 调用
@AutoWired
private ProductClient productClient;

productClient.productMsg()    // 调用到 product client中的服务
