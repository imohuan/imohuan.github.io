AuthenticationHandler

@Component
public class AppSuccessHandler implements AuthenticationSuccessHandler {
    @Autowired
    private ObjectMapper objectMapper;
    @Autowired
    private ClientDetailsService clientDetailsService;
    @Autowired
    private AuthorizationServerTokenServices defaultAuthorizationServerTokenServices;
     @Override
    public void onAuthenticationSuccess(HttpServletRequest request, HttpServletResponse        response, Authentication authentication) throws IOException, ServletException {

        // 获取头信息中authorization    具体代码 BasicAuthenticationFilter
        String header = request.getHeader("Authorization");

        if (header == null || !header.toLowerCase().startsWith("basic ")) {
            throw new UnapprovedClientAuthenticationException("Authorization不存在");
        }
        //一个方法解析为字符串数组格式
        String[] tokens = extractAndDecodeHeader(header, request);    
        assert tokens.length == 2;

        String clientId = tokens[0];
        String clientSecret = tokens[1];

        // 通过ClientId创建clientDetails clientDetailsService通过Autowired
        ClientDetails details = clientDetailsService.loadClientByClientId(clientId);    
        if (details == null) {
            throw new UnapprovedClientAuthenticationException("ClientId对应信息错误");
        } else if (!StringUtils.equals(details.getClientSecret(), clientSecret)) {
            throw new UnapprovedClientAuthenticationException("ClientSecret错误");
        }

        // 创建TokenRequest， 1. 不需要参数，应为我们需要authentication，但是SuccessHandler就已经帮我们创建了 2. ClientId 3. Scope 4.对应的授权方式（自定义）
        TokenRequest tokenRequest = new TokenRequest(Collections.emptyMap(), clientId, details.getScope(), "custom");
        // 创建OAuth2Request ，通过tokenRequest中放入clientDeatils
        OAuth2Request oAuth2Request = tokenRequest.createOAuth2Request(details);
        // 创建OAuth2Authentication，需要OAuth2Request和authentication
        OAuth2Authentication oAuth2Authentication = new OAuth2Authentication(oAuth2Request, authentication);
        // 创建出token
        OAuth2AccessToken accessToken = defaultAuthorizationServerTokenServices.createAccessToken(oAuth2Authentication);
        response.setContentType("application/json;charset=utf-8");
        response.getWriter().write(objectMapper.writeValueAsString(accessToken));
    }
}

JWT替换token

JWT特点
  1. JWT令牌标准，自包含（有意义信息） SpringSecurity默认是使用UUID无意义令牌
  2. 密签（签名）
  3. 可扩展
替换
// 此类放在Redis中
@Configuration
// 当配置properties前缀为prefix名称name是否为havingValue 如果是下面配置生效反则不生效，matchIfMissing代表没有配置也生效
@ConditionalOnProperty(prefix = "imohuan.security.oauth2", name = "store", havingValue = "jwt", matchIfMissing = true)
public static class Jwt {

    @Bean
    public TokenStore jwtTokenStore() {     //JWT的配置存储
        return new JwtTokenStore(jwtAccessTokenConverter());
    }

    @Bean
    public JwtAccessTokenConverter jwtAccessTokenConverter() {    //JWT生成规范
        JwtAccessTokenConverter jwtAccessTokenConverter = new JwtAccessTokenConverter();
        jwtAccessTokenConverter.setSigningKey("imohuan");    //设置密钥
        return jwtAccessTokenConverter;
    }
}

// 自动装配 因为不一定存在这个类，如果存在就知道我们指定了jwt的令牌模式
@Autowired(required = false)
private JwtAccessTokenConverter jwtAccessTokenConverter;

// configure(AuthorizationServerEndpointsConfigurer endpoints) 方法
if (jwtAccessTokenConverter != null) {
     endpoints.accessTokenConverter(jwtAccessTokenConverter);
}
配置JwtTokenEnhancer （token附加信息） 增强器
------- 创建自定义TokenEnhancer类
public class JwtTokenEnhancer implements TokenEnhancer {

    @Override
    public OAuth2AccessToken enhance(OAuth2AccessToken oAuth2AccessToken, OAuth2Authentication oAuth2Authentication) {

        // 向token添加一段信息
        Map<String, Object> info = new HashMap<>();
        info.put("hello", "欢迎来到我的世界");
        // 设置自定义附加信息
        ((DefaultOAuth2AccessToken) oAuth2AccessToken).setAdditionalInformation(info);

        return oAuth2AccessToken;
    }
}


------- 配置bean static class中
@Bean
@ConditionalOnMissingBean(name = "jwtTokenEnhancer")
public TokenEnhancer jwtTokenEnhancer() {
    return new JwtTokenEnhancer();
}

------- endpoints 方法 http配置
@Autowired(required = false)
private TokenEnhancer jwtTokenEnhancer;
@Autowired(required = false)
private JwtAccessTokenConverter jwtAccessTokenConverter;

if (jwtAccessTokenConverter != null && jwtTokenEnhancer != null) {
            TokenEnhancerChain tokenEnhancerChain = new TokenEnhancerChain();
            List<TokenEnhancer> tokenEnhancers = new ArrayList<>();
            tokenEnhancers.add(jwtTokenEnhancer);
            tokenEnhancers.add(jwtAccessTokenConverter);
            // 向TokenEnhancer链中添加tokenEnhancer 因为默认是DefaulOAuth2AccessToken
            // 创建的以UUID为token的值，因为是private方法，所以是固定的
            // 我们需要吧token变成jwt的方式，和增加自己的TokenEnhancer
            tokenEnhancerChain.setTokenEnhancers(tokenEnhancers);

            endpoints
                    .tokenEnhancer(tokenEnhancerChain)    // 设置增强链
                    .accessTokenConverter(jwtAccessTokenConverter);
}

SSO: 单点登陆

​ 比如说你登陆淘宝，天猫也会登陆，淘宝天猫是2个服务器，登陆界面也是一个服务器，所以登陆的时候我们是访问的认证服务器。认证服务器会给出一个token，在淘宝上就登陆了。

// 用户配置
@EnableOAuth2Sso    // 用户启动项配置
security.oauth2.client.access-token-uri=http:localhost:9999/server/oauth/token
security.oauth2.client.user-authorization-uri=http:localhost:9999/server/oauth/authorize
security.oauth2.resource.jwt.key-uri=http:localhost:9999/server/oauth/token_key

@Configuration
@EnableAuthorizationServer    // 配置认证服务器
public class AuthorizationServerConfig extends AuthorizationServerConfigurerAdapter {

    @Override
    public void configure(AuthorizationServerSecurityConfigurer security) throws Exception {
        security.tokenKeyAccess("isAuthenticated()");
    }

    @Override
    public void configure(ClientDetailsServiceConfigurer clients) throws Exception {
        clients.inMemory()
                .withClient("imohuan1")
                .secret("imohuansecret1")
                .authorizedGrantTypes("authorization_code", "refresh_token")
                .scopes("all")
                .and()
                .withClient("imohuan2")
                .secret("imohuansecret2")
                .authorizedGrantTypes("authorization_code", "refresh_token")
                .scopes("all");
    }

    @Override
    public void configure(AuthorizationServerEndpointsConfigurer endpoints) throws Exception {
        endpoints.tokenStore(jwtTokenStore())
                .accessTokenConverter(jwtAccessTokenConverter());
    }

    @Bean
    public TokenStore jwtTokenStore() {
        return new JwtTokenStore(jwtAccessTokenConverter());
    }

    @Bean
    public JwtAccessTokenConverter jwtAccessTokenConverter() {
        JwtAccessTokenConverter jwtAccessTokenConverter = new JwtAccessTokenConverter();
        jwtAccessTokenConverter.setSigningKey("imohuan");
        return jwtAccessTokenConverter;
    }

}
